93inline _scallop_read_t _scallop_read(
94 struct libadt_const_lptr script,
95 scallop_lang_lex_fn *const previous
96)
97{
98 wchar_t c = 0;
99 mbstate_t mbs = { 0 };
100 _scallop_read_t result = { 0 };
101 result.amount = _scallop_mbrtowc(&c, script, &mbs);
102 if (_scallop_read_error(result))
103 result.type = (scallop_lang_lex_fn*)scallop_lang_lex_unexpected;
104 else
105 result.type = (scallop_lang_lex_fn*)previous((wint_t)c);
106
107 result.script = libadt_const_lptr_index(script, (ssize_t)result.amount);
108 return result;
109}

146inline struct scallop_lang_token scallop_lang_token_next_raw(
147 struct scallop_lang_token previous
148)
149{
150 const ssize_t value_offset = (char *)previous.value.buffer
151 - (char *)previous.script.buffer;
152 struct libadt_const_lptr next = libadt_const_lptr_index(
153 previous.script,
154 value_offset + previous.value.length
155 );
156
157 _scallop_read_t
158 read = _scallop_read(next, previous.type),
159 previous_read = read;
160
161 if (_scallop_read_error(read))
162 return (struct scallop_lang_token) {
163 .script = previous.script,
164 .type = scallop_lang_lex_unexpected,
165 .value = libadt_const_lptr_truncate(next, 0),
166 };
167
168 if (read.type == scallop_lang_lex_end) {
169 return (struct scallop_lang_token) {
170 .script = previous.script,
171 .type = read.type,
172 .value = libadt_const_lptr_truncate(next, read.amount)
173 };
174 }
175
176 size_t value_length = read.amount;
177 for (
178 read = _scallop_read(read.script, read.type);
179 !_scallop_read_error(read);
180 read = _scallop_read(read.script, read.type)
181 ) {
182 if (read.type != previous_read.type)
183 break;
184
185 previous_read = read;
186 value_length += read.amount;
187 }
188
189 return (struct scallop_lang_token) {
190 .script = previous.script,
191 .type = previous_read.type,
192 .value = libadt_const_lptr_truncate(next, value_length),
193 };
194}

219inline struct scallop_lang_token scallop_lang_token_next(
220 struct scallop_lang_token previous
221)
222{
223 struct scallop_lang_token result = scallop_lang_token_next_raw(
224 previous
225 );
226
227 const bool end = result.type == scallop_lang_lex_end
228 || result.type == scallop_lang_lex_unexpected;
229 if (end)
230 return result;
231
232 // This does a tonne more work than necessary but
233 // I'm lazy
234 struct scallop_lang_token next = scallop_lang_token_next(
235 result
236 );
237
238 const bool is_word = scallop_lang_lex_is_word(result.type)
239 && scallop_lang_lex_is_word(next.type);
240 if (is_word)
241 result.value.length += next.value.length;
242
243 const bool has_word_separator
244 = result.type == scallop_lang_lex_word_separator
245 || next.type == scallop_lang_lex_word_separator;
246 const bool has_statement_separator
247 = result.type == scallop_lang_lex_statement_separator
248 || next.type == scallop_lang_lex_statement_separator;
249 if (has_word_separator && has_statement_separator) {
250 result.type = scallop_lang_lex_statement_separator;
251 result.value.length += next.value.length;
252 }
253 return result;
254}

275inline ssize_t scallop_lang_token_normalize_word(
276 struct libadt_const_lptr word,
277 struct libadt_lptr out
278)
279{
280 size_t read_amount = 0;
281 ssize_t total_read_amount = 0;
282
283 scallop_lang_lex_fn
284 *current = (scallop_lang_lex_fn *)scallop_lang_lex_begin;
285 for (
286 ;
287 libadt_const_lptr_in_bounds(word);
288 word = libadt_const_lptr_index(word, (ssize_t)read_amount)
289 ) {
290 wchar_t c = 0;
291 mbstate_t state = { 0 };
292 read_amount = _scallop_mbrtowc(&c, word, &state);
293
294 const bool read_error = read_amount == (size_t)-1
295 || read_amount == (size_t)-2;
296
297 if (read_error)
298 return -1;
299
300 current = (scallop_lang_lex_fn *)current((wint_t)c);
301
302 const bool is_error_type = current == scallop_lang_lex_unexpected;
303
304 if (is_error_type)
305 return -1;
306
307 const bool skip_type = current == scallop_lang_lex_single_quote
308 || current == scallop_lang_lex_single_quote_end
309 || current == scallop_lang_lex_double_quote
310 || current == scallop_lang_lex_double_quote_end
311 || current == scallop_lang_lex_escape;
312
313 if (!skip_type) {
314 if (libadt_lptr_in_bounds(out)) {
315 libadt_lptr_memmove(out, libadt_const_lptr_truncate(word, read_amount));
316 out = libadt_lptr_index(out, (ssize_t)read_amount);
317 }
318 total_read_amount += (ssize_t)read_amount;
319 }
320 }
321
322 return total_read_amount;
323}
